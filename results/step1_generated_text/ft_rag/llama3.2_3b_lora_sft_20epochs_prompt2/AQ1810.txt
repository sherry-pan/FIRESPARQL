What is the best performing model benchmarking the ImageNet ReaL dataset in terms of Params metric?

Generated SPARQL:
 SELECT?model?model_lbl
     WHERE {
      ?dataset       a                orkgc:Dataset;
                      rdfs:label      ?dataset_lbl.
        FILTER (str(?dataset_lbl) = "ImageNet ReaL")
        {
          SELECT?model?model_lbl
          WHERE {
           ?dataset       a                orkgc:Dataset;
                            orkgp:HAS_DATASET      ?dataset;
                            rdfs:label              ?dataset_lbl.
           ?benchmark      orkgp:HAS_DATASET      ?dataset.
           ?cont         orkgp:HAS_BENCHMARK     ?benchmark;
                          orkgp:P31               ?paper;
                          rdfs:label              ?model_lbl.
           ?eval         orkgp:HAS_EVALUATION   ?benchmark;
                          orkgp:P31               ?paper.
            OPTIONAL {
             ?eval           orkgp:P32               ?metric.
             ?metric          orkgp:HAS_VALUE        ?value.
             ?value           orkgp:HAS_DATASET      ?dataset;
                                orkgp:HAS_METRIC        ?metric;
                                rdfs:label              ?metric_lbl.
              FILTER (str(?metric_lbl) = "Params")
            }
           ?cont       orkgp:P5003               ?value.
            BIND(xsd:float(?value) AS?value)
          }
          ORDER BY DESC(?value)
          LIMIT 1
        }
