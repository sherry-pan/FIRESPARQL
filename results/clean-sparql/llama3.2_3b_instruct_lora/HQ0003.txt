SELECT DISTINCT?method
     WHERE {
      ?paper      a            orkgc:Paper;
      ?paper      rdfs:label  ?paper_lbl.
       FILTER(REGEX(?paper_lbl, "^[^[:space:]]*Knowledge representation method.*"))
      ?cont      orkgp:HAS_PAPER      ?paper.
      ?cont      orkgp:P31           ?method.
     }     ORDER BY DESC(?cont) LIMIT 1;
