SELECT?code
     WHERE {
      ?dataset       a                orkgc:Dataset;
                      rdfs:label      ?dataset_lbl.
       FILTER (str(?dataset_lbl) = "XLNet-Large")
      ?benchmark      orkgp:HAS_DATASET      ?dataset.
      ?cont          orkgp:HAS_BENCHMARK     ?benchmark.
      ?cont          orkgp:P31               ?code.
      ?code         rdfs:label              ?code_lbl.
       FILTER (str(?code_lbl) = "code")
     }
