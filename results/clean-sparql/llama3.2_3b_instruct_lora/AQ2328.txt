SELECT?code
     WHERE {
      ?model        a            orkgc:Model;
                   rdfs:label      ?model_lbl.
       FILTER (str(?model_lbl) = "MFEC")
      ?benchmark      orkgp:HAS_DATASET       ?dataset.
      ?cont           orkgp:HAS_BENCHMARK     ?benchmark.
      ?cont           orkgp:HAS_MODEL         ?model;
                       orkgp:HAS_SOURCE_CODE   ?code.
     } 
     END;
