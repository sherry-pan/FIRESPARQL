```sparql
PREFIX orkg: <http://www.orkg.org/>
PREFIX wos: <http://www.orkg.org/wos/>

WHERE {
 ?paper a orkg:ResearchPaper.
 ?paper wos:hasTitle ?title.
 ?paper wos:hasID ?id.
 ?paper wos:performedBenchmarkOn ?benchmark.

}
```
