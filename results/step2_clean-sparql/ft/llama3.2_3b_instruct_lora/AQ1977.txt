```sparql
SELECT ?code
WHERE {
  ?model a orkgc:Model;
         rdfs:label ?model_lbl.
  FILTER (str(?model_lbl) = "STREET")
  ?cont orkgp:HAS_MODEL ?model;
        orkgp:HAS_SOURCE_CODE ?code;
        orkgp:HAS_EVALUATION ?eval.
  ?eval rdfs:label ?eval_lbl.
  FILTER (str(?eval_lbl) = "Code Reference")
}
```
