```sparql
SELECT DISTINCT ?model ?model_lbl
WHERE {
  ?dataset a orkgc:Dataset;
           rdfs:label ?dataset_lbl.
  FILTER (str(?dataset_lbl) = "Atari 2600 Montezuma's Revenge" || str(?dataset_lbl) = "Atari 2600 Montezuma's Revenge (Test)")
  
  ?benchmark orkgp:HAS_DATASET ?dataset;
             orkgp:HAS_EVALUATION ?eval.
             
  ?eval orkgp:HAS_VALUE ?value.
  
  OPTIONAL {
    ?eval orkgp:HAS_METRIC ?metric.
    ?metric orkgp:HAS_PROPERTY ?property;
            orkgp:HAS_VALUE ?value.
  }
  
  ?cont orkgp:HAS_BENCHMARK ?benchmark;
        orkgp:HAS_METHOD ?method.
        
  OPTIONAL {
    ?method orkgp:HAS_IMPLEMENTATION ?implementation.
    ?implementation orkgp:HAS_APPROACH ?approach.
  }
  
  ?model orkgp:HAS_IMPLEMENTATION ?implementation.
  ?model rdfs:label ?model_lbl.
  
  FILTER(REGEX(STR(?value), "Average Return (NoOp)", "i"))
}
```
