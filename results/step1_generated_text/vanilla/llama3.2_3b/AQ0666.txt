     WHERE {
      ?paper orkg:hasBenchmark?benchmark.

      ?dataset orkg:name "ObjectNet"@en.
      ?benchmark orkg:hasPaper?paper.
      ?paper orkg:title?title.
      ?paper orkg:paperId?paperId.
     } 
     GROUP BY?paper?title?paperId
     HAVING COUNT(?benchmark) > 0 
     ORDER BY DESC (COUNT(?benchmark))
