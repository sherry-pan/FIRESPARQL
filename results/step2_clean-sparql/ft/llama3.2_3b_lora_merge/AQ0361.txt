```sparql
SELECT ?paper ?paper_lbl ?paper_id
WHERE {
    ?dataset a orkgc:Dataset;
             rdfs:label "SciERC".
    ?benchmark orkgp:HAS_DATASET ?dataset.
    ?cont orkgp:HAS_BENCHMARK ?benchmark.
    ?paper orkgp:P31 ?cont;
           rdfs:label ?paper_lbl;
           orkgc:ID ?paper_id.
}
LIMIT 10
```
