SELECT?location
     WHERE {
      ?location      a            orkgc:Location;
                      rdfs:label      ?location_lbl.
       FILTER (str(?location_lbl) = "most common")
      ?cont       orkgp:HAS_LOCATION      ?location.
     } 
     LIMIT 1; 
     ORDER BY DESC(?cont); 
     OFFSET 0; 
     LIMIT 1;
