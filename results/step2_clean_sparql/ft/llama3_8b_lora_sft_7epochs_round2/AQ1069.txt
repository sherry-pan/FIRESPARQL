```sparql
SELECT DISTINCT ?metric ?metric_lbl (MAX(?value) AS ?score)
WHERE {
  ?dataset a orkgc:Dataset;
           rdfs:label ?dataset_lbl.
  FILTER (str(?dataset_lbl) = "seel.cse.lsu.edu/data/re17.zip")
  ?benchmark orkgp:HAS_DATASET ?dataset;
             orkgp:HAS_EVALUATION ?eval.
  OPTIONAL {
    ?eval orkgp:HAS_METRIC ?metric.
    ?metric rdfs:label ?metric_lbl.
  }
  BIND(
    IF(?metric = ?_metric1, ?value1, 
    IF(?metric = ?_metric2, ?value2, 
    IF(?metric = ?_metric3, ?value3, ?value0)))
    AS ?value
  )
  ?cont orkgp:HAS_BENCHMARK ?benchmark.
  ?cont orkgp:HAS_EVALUATION ?eval_to_add.
  OPTIONAL {
    ?eval_to_add orkgp:HAS_METRIC ?_metric1.
    ?cont orkgp:HAS_VALUE ?value1.
  }
  OPTIONAL {
    ?eval_to_add orkgp:HAS_METRIC ?_metric2.
    ?cont orkgp:HAS_VALUE ?value2.
  }
  OPTIONAL {
    ?eval_to_add orkgp:HAS_METRIC ?_metric3.
    ?cont orkgp:HAS_VALUE ?value3.
  }
}
ORDER BY DESC(?score)
```
