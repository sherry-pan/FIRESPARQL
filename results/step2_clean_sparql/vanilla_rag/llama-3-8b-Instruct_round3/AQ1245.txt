```sparql
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#>
PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
PREFIX ex: <http://example.org/>

SELECT ?metric ?score
WHERE {
  ?result ex:dataset ex:AAPD .
  ?result ex:metric ?metric .
  ?result ex:score ?score .
}
ORDER BY DESC(?score)
LIMIT 1
```
