Provide a list of papers that have utilized the AlexNet, MultiGrasp model and include the links to their code?

 PREFIX orkg: <http://orkg.org/schema/2.0/>
     PREFIX wot: <http://www.wotontology.org/>
     PREFIX rdfs: <http://www.wotontology.org/rdfs/>
     PREFIX schema: <http://schema.org/>
     SELECT?paper?codeLink
     WHERE {
      ?paper wot:utilizes?model.
      ?model schema:hasCode?code.
       FILTER (lang(?paper) = 'en' AND lang(?code) = 'en')
       BIND (CONCAT('https://github.com/',?paper) AS?paperLink)
       BIND (CONCAT('https://github.com/',?code) AS?codeLink)
       FILTER (CONTAINS(LANG(?paper), 'AlexNet') AND CONTAINS(LANG(?code), 'MultiGrasp'))
     }
