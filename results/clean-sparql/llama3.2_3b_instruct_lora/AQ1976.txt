SELECT?code
     WHERE {
      ?papers rdfs:label?papers.
       FILTER (str(?papers) = "benchmarking")
      ?papers orkgp:P32?papers_p32.
      ?papers_p32 rdfs:label?papers_p32.
      ?papers_p32 orkgp:P33?code.
     }     CONTAIN?papers. 
     FILTER (str(?papers_p32) = "SEE")
     ORDER BY DESC(?papers_p32)
     LIMIT 10
