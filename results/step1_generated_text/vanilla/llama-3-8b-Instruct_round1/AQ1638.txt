Question:
Which model has achieved the highest Score score on the Ball in cup, catch (DMControl100k) benchmark dataset?

Generated SPARQL:
 PREFIX orkg: <http://www.orkg.org/>
     PREFIX owl: <http://www.w3.org/2002/07/owl#>
     PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#>
     SELECT?model?score
     WHERE {
      ?model a orkg:Model.
      ?model orkg:hasBenchmark?benchmark.
      ?benchmark rdfs:label "Ball in cup, catch (DMControl100k)".
      ?model orkg:hasScore?score.
      ?score a owl:Quantity.
      ?score orkg:unit "Score".
      ?model orkg:hasBestScore?bestScore.
      ?bestScore owl:equal?score.
       FILTER (?bestScore >?score)
     }<|eot_id|>
