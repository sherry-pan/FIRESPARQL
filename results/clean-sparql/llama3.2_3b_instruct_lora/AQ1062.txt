```sparql
SELECT ?metric ?metric_lbl (MAX(?value) AS ?max_value)
WHERE {
  ?dataset a orkgc:Dataset;
           rdfs:label ?dataset_lbl.
  FILTER (str(?dataset_lbl) = "Automatically labeled Medline abstracts corpus")
  
  ?benchmark orkgp:HAS_DATASET ?dataset;
             orkgp:HAS_EVALUATION ?eval.

  ?eval orkgp:HAS_VALUE ?value.
  
  ?metric orkgp:HAS_BENCHMARK ?benchmark;
          orkgp:HAS_METRIC ?metric_lbl.
  
  ?metric rdfs:label ?metric_lbl.
}
GROUP BY ?metric ?metric_lbl
ORDER BY DESC(?max_value)
LIMIT 1
```
