What is the highest benchmark result achieved on the Yelp-14 dataset, including the metric and its value?

SELECT?metric?metric_lbl (MAX(?value) AS?value_lbl)
WHERE {
 ?dataset       a                orkgc:Dataset;
                  rdfs:label      ?dataset_lbl.
  FILTER (str(?dataset_lbl) = "Yelp-14")
 ?benchmark      orkgp:HAS_DATASET      ?dataset.
 ?cont           orkgp:HAS_BENCHMARK     ?benchmark.
 ?eval           orkgp:P31               ?cont;
                  orkgp:HAS_EVALUATION   ?eval.
  OPTIONAL {?eval           orkgp:HAS_METRIC        ?metric.
          ?metric          rdfs:label              ?metric_lbl.}
 ?value          orkgp:HAS_EVALUATION_VALUE?eval.
  OPTIONAL {?value         orkgp:HAS_METRIC        ?metric.
          ?metric          rdfs:label              ?metric_lbl.}
}
ORDER BY DESC(?value)
LIMIT 1;
