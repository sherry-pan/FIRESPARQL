```sparql
SELECT ?dataset ?dataset_lbl
WHERE {
    ?problem a orkgc:Problem;
             rdfs:label "Common Sense Reasoning".
    ?dataset a orkgc:Dataset;
             rdfs:label ?dataset_lbl.
    ?benchmark orkgp:HAS_DATASET ?dataset.
    ?cont orkgp:HAS_BENCHMARK ?benchmark;
          orkgp:P32 ?problem.
}
```
