SELECT DISTINCT ?paper ?paper_lbl
WHERE {
    ?dataset a orkgc:Dataset;
             rdfs:label ?dataset_lbl.
    FILTER (str(?dataset_lbl) = "Amazon-2")
    ?benchmark orkgp:HAS_DATASET ?dataset;
               orkgp:P32 ?cont.
    ?cont orkgp:HAS_BENCHMARK ?paper.
    ?paper a orkgp:Paper;
           rdfs:label ?paper_lbl.
}
