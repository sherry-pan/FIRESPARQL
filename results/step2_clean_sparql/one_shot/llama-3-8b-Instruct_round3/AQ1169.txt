```sparql
SELECT ?score ?metric
WHERE {
  ?dataset a orkgc:Dataset;
           rdfs:label ?dataset_lbl.
  FILTER (str(?dataset_lbl) = "Words in Context")
  ?benchmark orkgp:HAS_DATASET ?dataset;
             orkgp:HAS_EVALUATION ?eval.
  ?eval orkgp:HAS_SCORE ?score;
         orkgp:HAS_METRIC ?metric.
}
```
